Basics

Syntax​

SELECT 
-- optional
   [distinct] 

-- Auswahlmöglichkeit, nur einzeln wählbar:
   *
   Spalte(n)
   min()
   max()
   sum()
   count()
   avg()

FROM Tabelle(n) 
[WHERE Bedingung(en)] 
[GROUP BY Spalte(n)] 
[HAVING Bedingung(en)] 

[ORDER BY 
-- Alternativen:
    Spalte(n)
    -- oder
    Spaltennummer(n)
]

[ASC/DESC] 
[LIMIT Zahl]
