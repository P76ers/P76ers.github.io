    }

    ```json title='settings.json'
    {
    "tag" : "HTML",
    "iconColor" : "orange",
    "highlightColor" : "#191919",
    "scope" : "workspace",
    "backgroundCoIor" : "#191919",
    "styleComment" : true,
    "borderStyle" : "1px solid #191919",
    "borderRadius" : 6
    }
    ```
    
  - usage Sections

    ```html title='example.html'
    <!-- SECTION One -->
      <!-- NOTE Nested -->
    <!-- !SECTION One -->
    ```

    ```js title='example.js'
    /* SECTION One */
      /* NOTE Nested */
    /* !SECTION One  */
    ```

<CardChapter>

  ### Prettier

<Section>
  <Card
    title='Marketplace'
    description='Download'
    to='https://marketplace.visualstudio.com/items?itemName=esbenp.prettier-vscode'
  />
  <Card
    title='Prettier'
    description='Homepage'
    to='https://prettier.io/'
  />
  <Card
    title='Prettier'
    description='Documentation'
    to='https://prettier.io/docs/en/index.html'
  />
  <CardInfo
    title='Info:'
    icon={<MyInfoIcon />}
    description='Code Formatter.'
  />
</Section>


  - Format Document on Save (general VsCode-Setting)

    ```json title='settings.json'
    "editor.formatOnSave": true,
    ```
  - set as default Formatter

    ```json title='settings.json'
    "editor.defaultFormatter": "esbenp.prettier-vscode",
    "[html]": {
      "editor.defaultFormatter": "esbenp.prettier-vscode"
    },
    "[css]": {
      "editor.defaultFormatter": "esbenp.prettier-vscode"
    },
    "[javascript]": {
      "editor.defaultFormatter": "esbenp.prettier-vscode"
    },
    "[json]": {
      "editor.defaultFormatter": "esbenp.prettier-vscode"
    },
    "[javascriptreact]": {
      "editor.formatOnSave": true,
      "editor.defaultFormatter": "esbenp.prettier-vscode"
    },
    "[typescript]": {
      "editor.defaultFormatter": "esbenp.prettier-vscode"
    },
    ```

  - Word wrap lenght

    ```json title='settings.json'
    "prettier.printWidth": 120,
    ```

  - Ignoring Files
    ```.prettierignore title='.prettierignore'
    *.mdx
    ```

</CardChapter>
