{
  T temp = a;
  a = b;
  b = temp;
}

{
  int intA = 5, intB = 10;
  double doubleA = 3.14, doubleB = 2.71;
  char charA = 'A', charB = 'B';

  swapValues(doubleA, doubleB);

  // Tausche zwei Char
